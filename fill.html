Matrix filler

Parameters:

  • a must be an Array.
    (matrix pointer)

  • v must be an any.
    (value used to fill the matrix)

  • dim must be an Array of ints.
    (list of the size of the different dimensions)

  • i must be an int.
    (current index in dim)

  • last must be an int.
    (last index to consider in dim)

var fill = function ( a , v , dim , i , last ) { var n , a , j ; n = dim[i] ; if ( i === last ) { for ( j = 0 ; j < n ; ++j ) { a[j] = v ; } } else { ++i ; for ( j = 0 ; j < n ; ++j ) { fill( a[j] , v , dim , i , last ); } } return a ; } ; exports.fill = fill ;
